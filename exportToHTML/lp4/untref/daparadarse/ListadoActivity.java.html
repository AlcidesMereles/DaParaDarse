216      public String leer(){ 
217          HttpClient cliente =new DefaultHttpClient(); 
218          HttpContext contexto = new BasicHttpContext(); 
219          HttpGet httpget = new HttpGet("http://daparadarse.site88.net/selectAllJSON.php"); 
220          String resultado=null; 
221          try { 
222              HttpResponse response = cliente.execute(httpget,contexto); 
223              HttpEntity entity = response.getEntity(); 
224              resultado = EntityUtils.toString(entity, "UTF-8"); 
225          } catch (Exception e) { 
226             e.printStackTrace(); 
227          } 
228          return resultado; 
229      }

256      public ArrayList<String> obtDatosJSON(String response){ 
257          ArrayList<String> listado= new ArrayList<String>(); 
258          try { 
259              JSONObject jsonObject = new JSONObject(response); 
260              JSONArray json = jsonObject.getJSONArray("personas"); 
261              String texto=""; 
262              for (int i=0; i<json.length();i++){ 
263                  texto = json.getJSONObject(i).getString("nombre") +" - "+ 
264                          json.getJSONObject(i).getString("apellido") +" - "+ 
265                          json.getJSONObject(i).getString("edad") +" - "+ 
266                          json.getJSONObject(i).getString("modo"); 
267                  listado.add(texto); 
268              } 
269          } catch (Exception e) { 
270  // TODO: handle exception 
271          } 
272          return listado; 
273      }
