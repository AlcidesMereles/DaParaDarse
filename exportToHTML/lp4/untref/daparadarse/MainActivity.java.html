62                   } 
63               } 
64           };

73       public String enviarPost(String nombre, String apellido, String edad) { 
74    
75           HttpClient httpClient = new DefaultHttpClient(); 
76           HttpContext localContext = new BasicHttpContext(); 
77           HttpPost httpPost = new HttpPost( 
78                   "http://www.daparadarse.site88.net/Android/PutData.php"); 
79           HttpResponse response = null; 
80           try { 
81               List<NameValuePair> params = new ArrayList<NameValuePair>(3); 
82               params.add(new BasicNameValuePair("nombre", nombre)); 
83               params.add(new BasicNameValuePair("apellido", apellido)); 
84               params.add(new BasicNameValuePair("edad", edad)); 
85               params.add(new BasicNameValuePair("modo", "POST")); 
86               httpPost.setEntity(new UrlEncodedFormEntity(params)); 
87               response = httpClient.execute(httpPost, localContext); 
88           } catch (Exception e) { 
89               // TODO: handle exception 
90           } 
91           return response.toString(); 
92    
93       }

95       public String enviarGet(String nombre, String apellido, String edad) { 
96           HttpClient httpClient = new DefaultHttpClient(); 
97           HttpContext localContext = new BasicHttpContext(); 
98           HttpResponse response = null; 
99           String parametros = "?nombre=" + nombre + "&apellido=" + apellido 
100                  + "&edad=" + edad + "&modo=GET"; 
101   
102          HttpGet httpget = new HttpGet( 
103                  "http://www.daparadarse.site88.net/Android/PutData.php" + parametros); 
104          try { 
105              response = httpClient.execute(httpget, localContext); 
106   
107          } catch (Exception e) { 
108   
109          } 
110          return response.toString(); 
111      }
